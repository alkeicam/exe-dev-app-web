Step 2.

Create post-commit file :

touch .git/hooks/post-commit

And paste into it following script :

#!/bin/sh
GIT_LOG=`git log --stat -1 HEAD | base64`
GIT_DIFF=`git show --unified | base64`
REMOTE=`git config --get remote.origin.url`
LOCAL=`git rev-parse --show-toplevel`
ACCOUNT="{{model.item.id}}"
USER="{{model.item.invitee.email}}"
PROJECT="{{model.item.project.id}}"

FINAL_REMOTE=""

if [ -n "$REMOTE" ]; then
    FINAL_REMOTE=$REMOTE
else
    FINAL_REMOTE=$LOCAL
fi

POST_BODY="{\"gitlog\":\"$GIT_LOG\",\"oper\":\"commit\",\"remote\":\"$FINAL_REMOTE\",\"diff\":\"$GIT_DIFF\",\"account\":\"$ACCOUNT\",\"user\":\"$USER\",\"project\":\"$PROJECT\"}"
curl -S -s -H "Content-Type: application/json" -d "$POST_BODY" http://localhost:5001/v1/hooks/commit

exit 0

Step 3.

Create pre-push file :

touch .git/hooks/pre-push

And paste into it following script :

#!/bin/sh

GIT_LOG=`git log --stat -1 HEAD | base64`
REMOTE=`git config --get remote.origin.url`
LOCAL=`git rev-parse --show-toplevel`
ACCOUNT="{{model.item.id}}"
USER="{{model.item.invitee.email}}"
PROJECT="{{model.item.project.id}}"

FINAL_REMOTE=""

if [ -n "$REMOTE" ]; then
    FINAL_REMOTE=$REMOTE
else
    FINAL_REMOTE=$LOCAL
fi

POST_BODY="{\"gitlog\":\"$GIT_LOG\",\"oper\":\"push\",\"remote\":\"$FINAL_REMOTE\",\"account\":\"$ACCOUNT\",\"user\":\"$USER\",\"project\":\"$PROJECT\"}"

curl -S -s -H "Content-Type: application/json" -d "$POST_BODY" http://localhost:5001/v1/hooks/push

exit 0

Step 4.

Make hooks executable :

chmod +x .git/hooks/post-commit && chmod +x .git/hooks/pre-push

Step 5.

Close this Guidebook and make some commits and pushes and check if the data is presented correctly.
